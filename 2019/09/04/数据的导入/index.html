1. 基于readr package

readr package中包括的函数主要有

read_csv()
read_csv2()
read_delim()
read_fwf()
read_tsv()

具体格式如下表

函数名称 细节
read_csv 分隔符号为逗号,write_file(x = “a,b,c\n1,2,3\n,4,5,NA”)
read_csv2 分隔符号为分号,write_file(x = “a;b;c\n1;2;3\n4;5;NA”)
read_delim 分隔符号为任意情况,read_delim(“file.txt”,delim = “]”)
read_fwf 读固定宽度的文件,read_fwf(‘file.fwf’,col position = c(1,3,5),write_file(x = “a b c\n1 2 3\n4 5 NA”))
read_tsv also read_table,阅读的是制表符类型的文件,write_file(x = “a\tb\tc\n1\t2\t3\n4\t5\tNA”)

函数里面的参数包括:

read_delim(file, delim, quote = "\"", escape_backslash = FALSE,
escape_double = TRUE, col_names = TRUE, col_types = NULL,
locale = default_locale(), na = c("", "NA"), quoted_na = TRUE,
comment = "", trim_ws = FALSE, skip = 0, n_max = Inf,
guess_max = min(1000, n_max), progress = show_progress(),
skip_empty_rows = TRUE)

read_csv(file, col_names = TRUE, col_types = NULL,
locale = default_locale(), na = c("", "NA"), quoted_na = TRUE,
quote = "\"", comment = "", trim_ws = TRUE, skip = 0,
n_max = Inf, guess_max = min(1000, n_max),
progress = show_progress(), skip_empty_rows = TRUE)

read_csv2(file, col_names = TRUE, col_types = NULL,
locale = default_locale(), na = c("", "NA"), quoted_na = TRUE,
quote = "\"", comment = "", trim_ws = TRUE, skip = 0,
n_max = Inf, guess_max = min(1000, n_max),
progress = show_progress(), skip_empty_rows = TRUE)

read_tsv(file, col_names = TRUE, col_types = NULL,
locale = default_locale(), na = c("", "NA"), quoted_na = TRUE,
quote = "\"", comment = "", trim_ws = TRUE, skip = 0,
n_max = Inf, guess_max = min(1000, n_max),
progress = show_progress(), skip_empty_rows = TRUE)

file指的是文件的路径,请使用引号,并且写明文件的全称,带文件格式,同时也可以接受字符串向量.
delim指的是单个用来起分隔作用的字符.
col_names接受三种情况的输入:
T,F,列名的字符串向量,
如果选择TRUE的话,那么第一行将被纳入为列名,
如果选择FALSE的话,那么将新建每一列的列名,
如果选择的为列名的字符串向量,那么形成的数据框的列名就为字符串向量
na参数指的是哪些字符串将被认为是缺失值,比如说na = c(“1”,”.”)
skip指的是在读入数据之前,有多少列应该被跳过
n_max指的是应该读入多少行的数据

2. 基于readxl package(.xlx,xlsx)

其中主要的函数及参数包括:

read_excel(path, sheet = NULL, range = NULL, col_names = TRUE,
col_types = NULL, na = "", trim_ws = TRUE, skip = 0,
n_max = Inf, guess_max = min(1000, n_max),
progress = readxl_progress(), .name_repair = "unique")

read_xls(path, sheet = NULL, range = NULL, col_names = TRUE,
col_types = NULL, na = "", trim_ws = TRUE, skip = 0,
n_max = Inf, guess_max = min(1000, n_max),
progress = readxl_progress(), .name_repair = "unique")

read_xlsx(path, sheet = NULL, range = NULL, col_names = TRUE,
col_types = NULL, na = "", trim_ws = TRUE, skip = 0,
n_max = Inf, guess_max = min(1000, n_max),
progress = readxl_progress(), .name_repair = "unique")

path指的是文件的路径,双引号,带文件的格式
sheet值得是阅读的哪个文件,不指定那么就默认为第一个文件
range指的是读取的范围,”B3:D87”这类的
col_names和col_types同上
na,skip,n_max都同上

3. 基于haven(SPSS,stata,SAS file)

其中的函数和主要参数包括

Usage

read_sav(file, encoding = NULL, user_na = FALSE)

read_por(file, user_na = FALSE)

write_sav(data, path, compress = FALSE)

read_spss(file, user_na = FALSE)

Arguments
file:Either a path to a file, a connection, or literal data (either a single string or a raw vector).
encoding:The character encoding used for the file. The default, NULL, use the encoding specified in the file, but sometimes this value is incorrect and it is useful to be able to override it.
user_na:If TRUE variables with user defined missing will be read into labelled_spss() objects. If FALSE, the default, user-defined missings will be converted to NA.
data:Data frame to write.
path:Path to a file where the data will be written.
compress:If TRUE, will compress the file, resulting in a .zsav file
